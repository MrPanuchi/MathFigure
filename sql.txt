CREATE TABLE PRODUCT (id INT IDENTITY(1,1) PRIMARY KEY, name NVARCHAR(20));
CREATE TABLE CATEGORY (id INT IDENTITY(1,1) PRIMARY KEY, name NVARCHAR(20));
CREATE TABLE PRODUCT_CATEGORY (id INT IDENTITY(1,1) PRIMARY KEY, id_p INT, id_c INT, FOREIGN KEY(id_p) REFERENCES PRODUCT(id), FOREIGN KEY(id_c) REFERENCES CATEGORY(id));
